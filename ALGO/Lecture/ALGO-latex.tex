    \begin{verbatim}
    Algorithm:

    int i, j, n, A[i][j], B[i][j], C[i][j];
    for(i = 0; i < n; i++){
        for(j = 0; j < n; j++)
            C[i][j] = A[i][j] + B[i][j];
    }
    
    space complexity                    Space complexity

    cost      repeat       total         cost  repeat  total
      1          1           1           i=4     1       4
    1+1+1    1+(n+1)+n      2n+2         j=4     1       4
    1+1+1   n+n(n+1)+n^2   2n^2+2n       n=4     1       4
      1         n^2         n^2         A[][]   4*n*n   4n^2
                                        B[][]   4*n*n   4n^2
                                        C[][]   4*n*n   4n^2

    F(n)= 3n^2 + 4n + 3                 S(n)= 12n^2 + 12
     TC -> O(n^2)                        SC -> O(n^2)
\end{verbatim}

\begin{verbatim}
    Algorithm:

    int p=0, i, n;
    for(i = 1; i <= p; i++)
        p+=1;

    Step analysis:
    i       p
    0      0+1
    1     0+1+2
    2    0+1+2+3
    3   0+1+2+3+4
    .       .
    .       .
    .       .
    k 0+1+2+3+4+...+k = k*(k+1)/2

    assume , p>n where step number is k and p = k*(k+1)/2
            k(k+1)/2 = n
         => k(k+1) = 2n
         => k^2 + k - 2n = 0
         => k^2 = 2n [ removed k as k^2 > k ]
         => k = sqrt(2n)
         => k = sqrt(2) * sqrt(n)

    thus O(sqrt(n))
\end{verbatim}

\end{verbatim}
